# THURSDAY CHALLENGES

## STRINGS

function greet(test) {
return "Hello " + test + "!";
}

## STRING LENGTH

function length(test) {
return test.length
}

let res = length("Gato")


## STRING: ToUpperCase()

function toCase(test) {
return test.toLowerCase() + "-" + test.toUpperCase()
}

let res = toCase("Gato")


## STRING: charAt()

function shortcut(first, last) {
return first.charAt(0) + last.charAt(0);
}

let res = shortcut("Amnesty", "International")
